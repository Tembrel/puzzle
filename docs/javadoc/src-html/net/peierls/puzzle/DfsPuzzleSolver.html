001package net.peierls.puzzle;
002
003import java.util.Optional;
004
005import java.util.function.Supplier;
006
007import one.util.streamex.StreamEx;
008
009
010/**
011 * Depth-first-search puzzle solver. Warning: This doesn't work very well unless
012 * the puzzle state can recognize hopelessness before the stack overflows.
013 */
014public class DfsPuzzleSolver<T extends PuzzleState<T>> extends FilteredPuzzleSolver<T> {
015
016    private final int maxDepth;
017
018    /**
019     * Constructs a DFS solver with an exact filter and the given max depth.
020     */
021    public DfsPuzzleSolver(int maxDepth) {
022        super();
023        this.maxDepth = maxDepth;
024    }
025
026    /**
027     * Constructs a DFS solver with the given max depth that will use
028     * a Bloom filter with the given parameters, otherwise an exact filter.
029     */
030    public DfsPuzzleSolver(int maxDepth, Supplier<PuzzleStateFilter<T>> filterSupplier) {
031        super(filterSupplier);
032        this.maxDepth = maxDepth;
033    }
034
035
036    @Override
037    protected Optional<T> solutionState(T initialState, PuzzleStateFilter<T> filter) {
038        if (initialState == null || filter == null) {
039            throw new NullPointerException("arguments must be non-null");
040        }
041        return dfs(initialState, filter, 0)
042            .findAny(PuzzleState::isSolution);
043    }
044
045    StreamEx<T> dfs(T state, PuzzleStateFilter<T> filter, int depth) {
046        state = filterState(state, filter);
047        if (state != null && depth < maxDepth) {
048            return StreamEx.of(state.successors())
049                .parallel()
050                .flatMap(next -> dfs(next, filter, depth + 1))
051                .prepend(state);
052        } else {
053            return StreamEx.empty();
054        }
055    }
056}
