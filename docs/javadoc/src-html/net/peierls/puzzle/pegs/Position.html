001package net.peierls.puzzle.pegs;
002
003import com.google.errorprone.annotations.Immutable;
004
005import java.util.Objects;
006
007/**
008 * Represents a grid position in a peg puzzle.
009 */
010@Immutable
011public class Position {
012    private final int row;
013    private final int col;
014
015    Position(int row, int col) { this.row = row; this.col = col; }
016
017    public int row() { return row; }
018    public int col() { return col; }
019
020    @Override public boolean equals(Object obj) {
021        if (this == obj) return true;
022        if (!(obj instanceof Position)) return false;
023        Position that = (Position) obj;
024        return this.row == that.row && this.col == that.col;
025    }
026    @Override public int hashCode() {
027        return Objects.hash(row, col);
028    }
029    @Override public String toString() {
030        return String.format("(%d, %d)", row, col);
031    }
032
033    public Position move(Position amount) {
034        return new Position(row + amount.row, col + amount.col);
035    }
036}
