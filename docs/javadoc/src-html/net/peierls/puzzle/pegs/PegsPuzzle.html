026/**
027 * A peg-jumping puzzle.
028 * The puzzle type holds common information about the puzzle, and the
029 * {@link PuzzleState} is an inner class of the puzzle type, to allow
030 * access to common information from state method implementations.
031 *
032 * @see <a href="https://www.chiark.greenend.org.uk/~sgtatham/puzzles/doc/pegs.html">Simon
033 * Tatham's puzzle page</a>
034 */
035public final class PegsPuzzle {
036
037    static class Move {
038        final JumpType jump;
039        final Position from;
040        Move(JumpType jump, Position from) {
041            this.jump = jump;
042            this.from = from;
043        }
044        Position from() { return from; }
045        Position to() { return jump.target(from); }
046        Position mid() { return jump.jumped(from); }
047        @Override public String toString() {
048            return String.format(
049                "Move peg at %s %s to %s, removing peg at %s",
050                from(), jump, to(), mid()
051            );
052        }
053    }
054
055    public final class State implements PuzzleState<State> {
056        private final State pred;
057        private final Move move;
058        private final BitSet pegs; // row major
059
060
061        State(Set<Position> pegs) {
062            this.pred = null;
063            this.move = null;
064            this.pegs = encoding.toRowMajor(pegs);
065        }
066
067        State(State pred, Move move) {
068            this.pred = pred;
069            this.move = move;
070            this.pegs = pred.applyJump(move);
071        }
072
073        @Override public boolean isSolution() {
074            return pegs.cardinality() == solutionCount;
075        }
076
077        @Override public Stream<State> successors() {
078            //System.out.println("Searching from " + this);
079            return encoding.legalJumps(rowMajorHoles, pegs)
080                .mapKeyValue(Move::new)
081                .map(move -> new State(this, move));
082        }
083
084        @Override public Optional<State> predecessor() {
085            return Optional.ofNullable(pred);
086        }
087
088        @Override public boolean equals(Object obj) {
089            if (this == obj) return true;
090            if (!(obj instanceof State)) return false;
091            State that = (State) obj;
092            return this.pegs.equals(that.pegs);
093        }
094
095        @Override public int hashCode() {
096            return Objects.hashCode(pegs);
097        }
098
099        @Override
100        public String toString() {
101            Set<Position> pegs = encoding.fromRowMajor(this.pegs);
102            StringBuilder buf = new StringBuilder();
103            if (move == null) {
104                buf.append("\nInitial position");
105            } else {
106                buf.append("\n").append(move);
107            }
108            buf.append(String.format(": #pegs=%d", pegs.size()));
109            rowMajorPositions(nrows, ncols).forEach(cur -> {
110                if (cur.col() == 0) {
111                    buf.append("\n");
112                }
113                String symbol = null;
114                if (move != null) {
115                    if (cur.equals(move.from())) {
116                        symbol = "-";
117                    } else if (cur.equals(move.to())) {
118                        symbol = "+";
119                    } else if (cur.equals(move.mid())) {
120                        symbol = "0";
121                    }
122                }
123                if (symbol == null) {
124                    if (pegs.contains(cur)) {
125                        symbol = "X";
126                    } else if (holes.contains(cur)) {
127                        symbol = ".";
128                    } else {
129                        symbol = " ";
130                    }
131                }
132                buf.append(symbol);
133            });
134            return buf.toString();
135        }
136
137        public Position dimensions() {
138            return new Position(nrows, ncols);
139        }
140
141        public Set<Position> holes() {
142            return holes;
143        }
144
145        private BitSet applyJump(Move move) {
146            BitSet newPegs = (BitSet) pegs.clone();
147            //System.out.printf("applying %s to %s:%n", move, pegs);
148            StreamEx.of(move.from(), move.to(), move.mid())
149                .mapToInt(encoding::toRowMajor)
150                //.peek(b -> System.out.printf("\t%d%n", b))
151                .forEach(newPegs::flip);
152            return newPegs;
153        }
154    }

157     * Pairs of jump offsets and corresponding common neighbors offsets.
158     */
159    private static ImmutableMap<Position, Position> JUMPS = ImmutableMap.of(
160        new Position(0, 2),    new Position(0, 1),
161        new Position(2, 0),    new Position(1, 0),
162        new Position(0, -2),   new Position(0, -1),
163        new Position(-2, 0),   new Position(-1, 0)
164    );

167    private final int nrows;
168    private final int ncols;
169    private final ImmutableSet<Position> holes;
170    private final ImmutableSet<Position> pegs;
171    private final int solutionCount;
172    private final PegEncoding encoding;
173    private final BitSet rowMajorHoles;
174
175
176    public PegsPuzzle(int nrows, int ncols, Set<Position> holes, Set<Position> pegs) {
177        this(nrows, ncols, holes, pegs, 1); // default solution is to get to 1 peg
178    }
179
180    public PegsPuzzle(int nrows, int ncols, Set<Position> holes, Set<Position> pegs, int solutionCount) {
181        if (!holes.stream().allMatch(hole -> hole.row() >= 0
182                                          && hole.row() < nrows
183                                          && hole.col() >= 0
184                                          && hole.col() < ncols)) {
185            throw new IllegalArgumentException(String.format("holes must all be in %d x %s grid", nrows, ncols));
186        }
187        if (!pegs.stream().allMatch(holes::contains)) {
188            throw new IllegalArgumentException("all pegs must be in holes");
189        }
190        this.nrows = nrows;
191        this.ncols = ncols;
192        this.holes = ImmutableSet.copyOf(holes);
193        this.pegs = ImmutableSet.copyOf(pegs);
194        this.solutionCount = solutionCount;
195        this.encoding = new PegEncoding(nrows, ncols);
196        this.rowMajorHoles = encoding.toRowMajor(holes);
197    }
198
199    public int nrows() { return nrows; }
200    public int ncols() { return ncols; }
201    public ImmutableSet<Position> holes() { return holes; }
202    public ImmutableSet<Position> pegs() { return pegs; }
203    public int solutionCount() { return solutionCount; }
204
205
206    public Optional<List<State>> solve(PuzzleSolver<State> solver) {
207        return solver.solution(new State(pegs));
208    }

210    /**
211     * Creates the initial state of a cross-shaped peg puzzle with the
212     * initial empty hole in the center. The containing square must have
213     * odd dimension, and the size of each arm must be odd and no greater
214     * than the side of the containing square.
215     * @param size number of holes in the center column (or center row)
216     * @param armSize number of columns or rows in each arm of the cross
217     * @throws IllegalArgumentException if {@code armSize > size}, if
218     * either is not odd and positive
219     */
220    public static PegsPuzzle makeCross(int size, int armSize, int solutionCount) {
221        int cornerSize = (size - armSize) / 2;
222        int nPegs = (size * size) - (cornerSize * cornerSize * 4) - 1;
223
224        //System.out.printf(
225        //    "Making cross puzzle of size %d, arm size %d, corner size %d, and #pegs %d%n",
226        //    size, armSize, cornerSize, nPegs);
227
228        Position center = new Position(size/2, size/2);
229        return new PegsPuzzle(size, size,
230            crossHoles(size, armSize).toSet(),
231            crossHoles(size, armSize).removeBy(c -> c, center).toSet(),
232            solutionCount
233        );
234    }
235
236    public static Funnel<State> stateFunnel() {
237        return (from, into) -> into.putBytes(from.pegs.toByteArray());
238    }
239
240    static StreamEx<Position> crossHoles(int size, int armSize) {
241        if (armSize > size || armSize < 1 || size < 1 || (armSize % 2) != 1 || (size % 2) != 1) {
242            throw new IllegalArgumentException(
243                "size arguments must be odd positive, arm size must be <= size");
244        }
245        int cornerSize = (size - armSize) / 2;
246        int minArm = cornerSize;
247        int maxArm = size - cornerSize - 1;
248        return rowMajorPositions(size, size)
249            .remove(c ->
250                (c.row() < minArm || c.row() > maxArm) &&
251                (c.col() < minArm || c.col() > maxArm)
252            );
253    }
254
255    static StreamEx<Position> rowMajorPositions(int nrows, int ncols) {
256        return IntStreamEx.range(0, nrows * ncols)
257            .mapToEntry(i -> i / ncols, i -> i % ncols)
258            .mapKeyValue((r, c) -> new Position(r, c));
259    }
260
261    public static void main(String[] args) {
262        int size = Integer.parseInt(args[0]);
263        int armSize = Integer.parseInt(args[1]);
264        int remainder = size - armSize;
265        int pegCount = size * size - remainder * remainder - 1;
266        int solutionCount = 1;
267        PegsPuzzle puzzle = makeCross(size, armSize, solutionCount);
268        PuzzleSolver<State> solver = new DfsPuzzleSolver<>(
269            //() -> new BloomPuzzleStateFilter<>(stateFunnel(), 100_000_000L, 0.0001)
270        );
271        System.out.printf("Solving %d x %d cross puzzle with arm %d, #pegs = %d, solution at %d%n",
272            size, size, armSize, pegCount, solutionCount);
273        Optional<List<State>> solution = puzzle.solve(solver);
274        String result = solution.map(s -> StreamEx.of(s).joining("\n")).orElse("no solution");
275        System.out.println(result);
276    }
