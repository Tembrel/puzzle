001package net.peierls.puzzle;
002
003import java.util.List;
004import java.util.Optional;
005
006
007/**
008 * Interface for algorithms that search for solutions to a puzzles
009 * of the parameterized type.
010 */
011public interface PuzzleSolver<T extends PuzzleState<T>> {
012
013    /**
014     * Finds a solution to puzzle defined by initial state, using
015     * the state filter supplied by that initial state.
016     *
017     * @return the solution to the puzzle as a list of states from
018     * the initial state to a solved state (inclusive), or an empty
019     * optional if solution is unreachable from initial state.
020     */
021    Optional<List<T>> solution(T initialState);
022}
