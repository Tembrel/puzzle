001package net.peierls.puzzle;
002
003import java.util.stream.*;
004
005import one.util.streamex.*;
006
007
008/**
009 * A state that a puzzle of parameterized type T can be in.
010 */
011public interface PuzzleState<T extends PuzzleState<T>> {
012
013    /**
014     * Returns whether this is a solution state.
015     */
016    boolean isSolution();
017
018    /**
019     * Returns true if it is known that no solution
020     * state can be reached from this state.
021     * Returns false otherwise, i.e., if it is not known
022     * that no solution state can be reached from this state,
023     * or if it is known that a solution state can be reached
024     * from this state.
025     */
026    boolean isHopeless();
027
028    /**
029     * The stream of puzzle states obtainable by
030     * a single move from this state.
031     */
032    Stream<T> successors();
033
034
035    /**
036     * Returns a state that is equivalent to this state,
037     * but which might implement some methods more efficiently,
038     * e.g., by precomputing commonly needed values.
039     * The return value can be this instance itself, and that is
040     * how the method is implemented by default.
041     */
042    @SuppressWarnings("unchecked")
043    default T initialized() {
044        return (T) this;
045    }
046
047
048    /**
049     * A rating of how good this state is. Lower is better. You can
050     * return the same value for all states if you don't know.
051     */
052    default int score() {
053        return 0;
054    }
055}
