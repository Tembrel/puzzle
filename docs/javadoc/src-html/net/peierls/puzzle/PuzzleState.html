001package net.peierls.puzzle;
002
003import java.util.Optional;
004import java.util.stream.Stream;
005
006
007/**
008 * A representation of the state of a puzzle that involves
009 * finding a sequence of moves from an initial state to
010 * a solution state.
011 * <ul>
012 * <li>
013 * A state can stream its successors, which are the states
014 * resulting in a single move from it.
015 * </li><li>
016 * A puzzle state can report whether it is a solution state or
017 * whether it is "hopeless", meaning that no sequence of
018 * moves from it will result in a solution state.
019 * </li><li>
020 * A puzzle state knows its predecessor state, and this
021 * information is not required to be used in defining state
022 * equivalence, i.e., two states with different predecessors are
023 * permitted to compare equal.
024 * </li><li>
025 * A puzzle state may be initialized with precomputed values through the
026 * {@link #initialized} method. This allows solvers to avoid unnecessary
027 * expensive computation on states that have already been seen.
028 * It is permissible to return self from this method if precautions
029 * for mutability under concurrent access are taken, e.g., if the state
030 * type is immutable and no precomputation is needed.
031 * </li><li>
032 * A puzzle state can produce a solution, i.e., a list of states from some
033 * initial state to the current state, where each element other than the
034 * first is a successor to the previous state in the list. The current state
035 * is not required to be a solution state.
036 * </li>
037 * </ul>
038 */
039public interface PuzzleState<T extends PuzzleState<T>> {
040
041    /**
042     * Returns whether this is a solution state.
043     */
044    boolean isSolution();
045
046    /**
047     * Returns true if it is known that no solution
048     * state can be reached from this state.
049     * Returns false otherwise, i.e., if it is not known
050     * that no solution state can be reached from this state,
051     * or if it is known that a solution state can be reached
052     * from this state. The default implementation returns false.
053     */
054    default boolean isHopeless() {
055        return false;
056    }
057
058    /**
059     * The stream of puzzle states obtainable by
060     * a single move from this state.
061     */
062    Stream<T> successors();
063
064
065    /**
066     * The state from which this state was reached,
067     * or empty if this is an initial state.
068     */
069    Optional<T> predecessor();
070
071
072    /**
073     * Returns a state that is equivalent to this state,
074     * but which might implement some methods more efficiently,
075     * e.g., by precomputing commonly needed values.
076     * The return value can be this instance itself, and that is
077     * how the method is implemented by default.
078     */
079    @SuppressWarnings("unchecked")
080    default T initialized() {
081        return (T) this;
082    }
083
084
085    /**
086     * A rating of how good this state is. Lower is better. You can
087     * return the same value for all states if you don't know.
088     * Puzzle solvers are not required to pay attention to scores
089     * unless they are looking for a best solution. The default
090     * implementation returns zero.
091     */
092    default int score() {
093        return 0;
094    }
095}
