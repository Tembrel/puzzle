001package net.peierls.puzzle;
002
003import com.google.common.hash.Funnel;
004
005import java.util.List;
006import java.util.Optional;
007import java.util.stream.*;
008
009import one.util.streamex.*;
010
011
012/**
013 * A representation of the state of a puzzle that involves
014 * finding a sequence of moves from an initial state to
015 * a solution state.
016 * <ul>
017 * <li>
018 * A state can stream its successors, which are the states
019 * resulting in a single move from it.
020 * </li><li>
021 * A puzzle state can report whether it is a solution state or
022 * whether it is "hopeless", meaning that no sequence of
023 * moves from it will result in a solution state.
024 * </li><li>
025 * A puzzle state knows its predecessor state, and this
026 * information is not required to be used in defining state
027 * equivalence, i.e., two states with different predecessors are
028 * permitted to compare as equal.
029 * </li><li>
030 * A puzzle state may be initialized with precomputed values through the
031 * {@link #precomputed} method. This allows solvers to avoid unnecessary
032 * expensive computation on states that have already been seen.
033 * It is permissible to return self from this method if precautions
034 * for mutability under concurrent access are taken, e.g., if the state
035 * type is immutable and no precomputation is needed.
036 * </li><li>
037 * A puzzle state type may optionally define a funnel, which if it exists
038 * can be used to produce a (potentially) lossy encoding of states of that type
039 * for use in approximate containment tests.
040 * </li><li>
041 * A puzzle state can produce a solution, i.e., a list of states from some
042 * initial state to the current state, where each element other than the
043 * first is a successor to the previous state in the list. The current state
044 * is not required to be a solution state.
045 * </li>
046 * </ul>
047 */
048public interface PuzzleState<T extends PuzzleState<T>> {
049
050    /**
051     * Returns whether this is a solution state.
052     */
053    boolean isSolution();
054
055    /**
056     * Returns true if it is known that no solution
057     * state can be reached from this state.
058     * Returns false otherwise, i.e., if it is not known
059     * that no solution state can be reached from this state,
060     * or if it is known that a solution state can be reached
061     * from this state.
062     */
063    default boolean isHopeless() {
064        return false;
065    }
066
067    /**
068     * The stream of puzzle states obtainable by
069     * a single move from this state.
070     */
071    Stream<T> successors();
072
073
074    /**
075     * The state from which this state was reached.
076     */
077    default Optional<T> predecessor() {
078        return Optional.empty();
079    }
080
081
082    /**
083     * Returns a state that is equivalent to this state,
084     * but which might implement some methods more efficiently,
085     * e.g., by precomputing commonly needed values.
086     * The return value can be this instance itself, and that is
087     * how the method is implemented by default.
088     */
089    @SuppressWarnings("unchecked")
090    default T precomputed() {
091        return (T) this;
092    }
093
094
095    /**
096     * A rating of how good this state is. Lower is better. You can
097     * return the same value for all states if you don't know.
098     * Puzzle solvers are not required to pay attention to scores
099     * unless they are looking for a best solution.
100     */
101    default int score() {
102        return 0;
103    }
104
105
106    /**
107     * Optionally returns a funnel for encoding arbitrary instances
108     * of this type, for use in approximate containment tests.
109     */
110    default Optional<Funnel<T>> funnel() {
111        return Optional.empty();
112    }
113
114    /**
115     * Convert this state into a solution.
116     */
117    default List<T> toSolution() {
118        @SuppressWarnings("unchecked")
119        List<T> reversed = StreamEx.iterate(
120            (T) this,
121            s -> s != null,
122            s -> s.predecessor().orElse(null)
123        ).toList();
124
125        return StreamEx.ofReversed(reversed).toList();
126    }
127
128}
