001package net.peierls.puzzle;
002
003import com.google.common.hash.Funnel;
004
005import java.util.List;
006import java.util.Optional;
007import java.util.stream.*;
008
009import one.util.streamex.*;
010
011
012/**
013 * A representation of the state of a puzzle that involves
014 * finding a sequence of moves from an initial state to
015 * a solution state.
016 * <ul>
017 * <li>
018 * A state can stream its successors, which are the states
019 * resulting in a single move from it.
020 * </li><li>
021 * A puzzle state can report whether it is a solution state or
022 * whether it is "hopeless", meaning that no sequence of
023 * moves from it will result in a solution state.
024 * </li><li>
025 * A puzzle state knows its predecessor state, and this
026 * information is not required to be used in defining state
027 * equivalence, i.e., two states with different predecessors are
028 * permitted to compare as equal.
029 * </li><li>
030 * There is no separate puzzle type, because an initial puzzle state
031 * instance essentially defines the puzzle: Find a sequence of states
032 * starting with the initial state and ending with a solution state
033 * with each element of the sequence other than first being a successor
034 * of the previous element of the sequence.
035 * </li><li>
036 * A puzzle state may be initialized with precomputed values through the
037 * {@link #precomputed} method. This allows solvers to avoid unnecessary
038 * expensive computation on states that have already been seen.
039 * It is permissible to return self from this method if precautions
040 * for mutability under concurrent access are taken, e.g., if the state
041 * type is immutable and no precomputation is needed.
042 * </li><li>
043 * A puzzle state type may optionally define a funnel, which if it exists
044 * can be used to produce a (potentially) lossy encoding of states of that type
045 * for use in approximate containment tests.
046 * </li><li>
047 * A puzzle state can produce a solution, i.e., a list of states from some
048 * initial state to the current state, where each element other than the
049 * first is a successor to the previous state in the list. The current state
050 * is not required to be a solution state.
051 * </li>
052 * </ul>
053 */
054public interface PuzzleState<T extends PuzzleState<T>> {
055
056    /**
057     * Returns whether this is a solution state.
058     */
059    boolean isSolution();
060
061    /**
062     * Returns true if it is known that no solution
063     * state can be reached from this state.
064     * Returns false otherwise, i.e., if it is not known
065     * that no solution state can be reached from this state,
066     * or if it is known that a solution state can be reached
067     * from this state.
068     */
069    boolean isHopeless();
070
071    /**
072     * The stream of puzzle states obtainable by
073     * a single move from this state.
074     */
075    Stream<T> successors();
076
077
078    /**
079     * The state from which this state was reached.
080     */
081    default Optional<T> predecessor() {
082        return Optional.empty();
083    }
084
085
086    /**
087     * Returns a state that is equivalent to this state,
088     * but which might implement some methods more efficiently,
089     * e.g., by precomputing commonly needed values.
090     * The return value can be this instance itself, and that is
091     * how the method is implemented by default.
092     */
093    @SuppressWarnings("unchecked")
094    default T precomputed() {
095        return (T) this;
096    }
097
098
099    /**
100     * A rating of how good this state is. Lower is better. You can
101     * return the same value for all states if you don't know.
102     * <strong>This value is currently ignored by the framework.</strong>
103     */
104    default int score() {
105        return 0;
106    }
107
108
109    /**
110     * Optionally returns a funnel for encoding arbitrary instances
111     * of this type, for use in approximate containment tests.
112     */
113    default Optional<Funnel<T>> funnel() {
114        return Optional.empty();
115    }
116
117    /**
118     * Convert this state into a solution.
119     */
120    default List<T> toSolution() {
121        @SuppressWarnings("unchecked")
122        List<T> reversed = StreamEx.iterate(
123            (T) this,
124            s -> s != null,
125            s -> s.predecessor().orElse(null)
126        ).toList();
127
128        return StreamEx.ofReversed(reversed).toList();
129    }
130
131}
