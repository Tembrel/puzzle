001package net.peierls.puzzle;
002
003import com.google.common.hash.Funnel;
004
005import java.util.List;
006import java.util.Optional;
007import java.util.function.Function;
008
009
010/**
011 * Abstract base for puzzle solvers that use a {@link PuzzleStateFilter}
012 * to keep track (possibility only approximately) of which states have
013 * been seen during the search for a solution. General-purpose solvers
014 * necessarily fall into this category, because otherwise there is no
015 * way to know in general whether a particular state has already been
016 * examined. (Custom solvers for specific puzzle state types might be
017 * able to get around this with domain-specific information.)
018 * <p>
019 * Because in general the search space can be so large that it is not
020 * possible to store all states that have been seen (even taking advantage
021 * of compressed forms), the {@link PuzzleStateFilter} abstraction is
022 * used to allow for searches that use much less memory but that may fail
023 * occasionally to search a valid branch.
024 */
025public abstract class AbstractPuzzleSolver<T extends PuzzleState<T>> implements PuzzleSolver<T> {
026
027    private final Function<Funnel<T>, PuzzleStateFilter<T>> funnelToFilter;
028
029
030    /**
031     * Constructs a solver that will always use an exact filter, which
032     * will store each state seen but will always report correctly on
033     * whether a state has been seen.
034     */
035    AbstractPuzzleSolver() {
036        this.funnelToFilter = funnel -> exactFilter();
037    }
038
039    /**
040     * Constructs a solver that will always use the given state filter.
041     */
042    AbstractPuzzleSolver(PuzzleStateFilter<T> filter) {
043        this.funnelToFilter = funnel -> filter;
044    }
045
046    /**
047     * Constructs a solver that will use an exact filter when solving
048     * for initial states that do not define a funnel, and a BloomFilter
049     * when solving initial states that do define a funnel. The two
050     * parameters to this method will be ignored in the former case.
051     */
052    AbstractPuzzleSolver(int expectedInsertions, double fpp) {
053        this.funnelToFilter = funnel -> bloomFilter(funnel, expectedInsertions, fpp);
054    }
055
056
057    @Override
058    public final Optional<List<T>> solution(T initialState) {
059        PuzzleStateFilter<T> filter = initialState.funnel()
060            .map(funnel -> funnelToFilter.apply(funnel))
061            .orElseGet(this::exactFilter);
062        return solution(initialState, filter);
063    }
064
065
066    /**
067     * Concrete subclasses must implement this method to
068     * find a solution from the given initial state, using
069     * the given state filter.
070     */
071    abstract Optional<List<T>> solution(T initialState, PuzzleStateFilter<T> filter);
072
073
074    private PuzzleStateFilter<T> bloomFilter(Funnel<T> funnel, int expectedInsertions, double fpp) {
075        return new BloomPuzzleStateFilter<>(funnel, expectedInsertions, fpp);
076    }
077
078    private PuzzleStateFilter<T> exactFilter() {
079        return new ExactPuzzleStateFilter<>();
080    }
081}
