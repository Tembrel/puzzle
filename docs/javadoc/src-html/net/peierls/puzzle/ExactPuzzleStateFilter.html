001package net.peierls.puzzle;
002
003import java.util.concurrent.*;
004
005/**
006 * An implementation of {@link PuzzleStateFilter} with exact containment.
007 * It has these properties:
008 * <ul>
009 * <li>
010 * {@link #mightContain} returns true <em>iff</em> the given
011 * state has been added;
012 * </li>
013 * <li>
014 * the return value of {@link #put} reflects whether the
015 * state was added for the first time;
016 * </li>
017 * <li>
018 * {@link #expectedFalsePositiveProbability} always returns {@code 0.0};
019 * and
020 * </li>
021 * <li>
022 * {@link #approximateElementCount} is always the true number of
023 * distinct elements added.
024 * </li>
025 * </ul>
026 */
027public class ExactPuzzleStateFilter<T extends PuzzleState<T>>
028        implements PuzzleStateFilter<T> {
029
030    private final ConcurrentMap<T, T> map = new ConcurrentHashMap<>();
031
032
033    @Override
034    public boolean mightContain(T state) {
035        return map.containsKey(state);
036    }
037
038    @Override
039    public boolean put(T state) {
040        return map.putIfAbsent(state, state) == null;
041    }
042
043    @Override
044    public long approximateElementCount() {
045        return map.size();
046    }
047
048    @Override
049    public double expectedFalsePositiveProbability() {
050        return 0.0;
051    }
052}
