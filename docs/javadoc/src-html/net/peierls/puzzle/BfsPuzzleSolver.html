001package net.peierls.puzzle;
002
003import java.util.Optional;
004
005import one.util.streamex.StreamEx;
006
007
008/**
009 * Breadth-first-search puzzle solver.
010 */
011public class BfsPuzzleSolver<T extends PuzzleState<T>> extends AbstractPuzzleSolver<T> {
012
013    /**
014     * Constructs a BFS solver with an exact filter.
015     */
016    public BfsPuzzleSolver() {
017        super();
018    }
019
020    /**
021     * Constructs a BFS solver that will use a Bloom filter for states
022     * that support it, otherwise an exact filter.
023     */
024    public BfsPuzzleSolver(int expectedInsertions, double fpp) {
025        super(expectedInsertions, fpp);
026    }
027
028
029    @Override
030    Optional<T> solutionState(T initialState, PuzzleStateFilter<T> filter) {
031        return bfs(StreamEx.of(initialState), filter)
032            //.peek(this::trace)
033            .findAny(PuzzleState::isSolution);
034    }
035
036    StreamEx<T> bfs(StreamEx<T> states, PuzzleStateFilter<T> filter) {
037        return states.headTail((state, rest) ->
038            bfs(maybeAddSuccessors(state, rest, filter), filter)
039                .prepend(state));
040    }
041
042    StreamEx<T> maybeAddSuccessors(T state, StreamEx<T> rest, PuzzleStateFilter<T> filter) {
043        if (filter.put(state) && !state.isHopeless()) {
044            // We haven't seen this state before and it isn't hopeless to search
045            // further from it, so we append its successors to whatever is left
046            // in the queue.
047            return rest.append(state.successors());
048        } else {
049            // Either we've seen this state before or it's hopeless to search
050            // its successors, so we just return whatever's left in the queue.
051            return rest;
052        }
053    }
054
055    private void trace(T state) {
056        System.out.printf("searching %s, pred %s%n",
057            state,
058            state.predecessor().map(Object::toString).orElse("-"));
059    }
060}
