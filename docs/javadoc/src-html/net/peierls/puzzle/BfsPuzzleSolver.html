001package net.peierls.puzzle;
002
003import java.util.ArrayDeque;
004import java.util.Deque;
005import java.util.Optional;
006
007import one.util.streamex.StreamEx;
008
009
010/**
011 * Breadth-first-search puzzle solver.
012 */
013public class BfsPuzzleSolver<T extends PuzzleState<T>> extends AbstractPuzzleSolver<T> {
014
015    /**
016     * Constructs a BFS solver with an exact filter.
017     */
018    public BfsPuzzleSolver() {
019        super();
020    }
021
022    /**
023     * Constructs a BFS solver that will use a Bloom filter for states
024     * that support it, otherwise an exact filter.
025     */
026    public BfsPuzzleSolver(int expectedInsertions, double fpp) {
027        super(expectedInsertions, fpp);
028    }
029
030
031    @Override
032    protected Optional<T> solutionState(T initialState, PuzzleStateFilter<T> filter) {
033        return bfs(initialState, filter)
034            //.peek(this::trace)
035            .findAny(PuzzleState::isSolution);
036    }
037
038    private StreamEx<T> bfs(T initialState, PuzzleStateFilter<T> filter) {
039        Deque<T> queue = new ArrayDeque<>();
040        queue.offer(initialState);
041        return StreamEx.produce(action -> {
042            T state = queue.poll();
043            if (state == null) {
044                return false;
045            }
046            state = filterState(state, filter);
047            if (state != null) {
048                action.accept(state);
049                state.successors().forEach(queue::offer);
050            }
051            return true;
052        });
053    }
054
055    private StreamEx<T> bfs2(T state, PuzzleStateFilter<T> filter) {
056        if (state == null) {
057            throw new NullPointerException();
058        }
059        return bfs_(filter, StreamEx.of(state));
060    }
061
062    private StreamEx<T> bfs_(PuzzleStateFilter<T> filter, StreamEx<T> states) {
063        return states.headTail((state, rest) ->
064            bfs_(filter, StreamEx.of(state.successors())
065                .map(next -> filterState(next, filter))
066                .nonNull()
067                .prepend(rest)
068            ).prepend(state)
069        );
070    }
071
072    private void trace(T state) {
073        System.out.printf("searching %s, pred %s%n",
074            state,
075            state.predecessor().map(Object::toString).orElse("-"));
076    }
077}
